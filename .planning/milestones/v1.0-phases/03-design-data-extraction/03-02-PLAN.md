export interface ColorToken {
  value: string;        // #RRGGBB or rgba(R,G,B,A)

  usageCount: number;
  nodeIds: string[];
  source: ('fill' | 'stroke' | 'shadow')[];
}

export interface GradientToken {
  value: string;        // CSS gradient syntax
  name: string;
  gradientType: string;
  usageCount: number;
  nodeIds: string[];
}

export interface TypographyToken {
  fontFamily: string;
  fontSize: number;
  fontWeight: number;
  lineHeight: number | null;
  letterSpacing: number;
  name: string;
  usageCount: number;
  nodeIds: string[];
}

export interface SpacingToken {
  value: number;
  usageCount: number;
  sources: string[];
}

export interface BorderToken {
  radius: number | null;
  cornerRadii: number[] | null;
  strokeColor: string | null;
  strokeWeight: number | null;
  name: string;
  usageCount: number;
  nodeIds: string[];
}

export interface ShadowToken {
  type: 'drop-shadow' | 'inner-shadow';
  color: string;
  offsetX: number;
  offsetY: number;
  blur: number;
  spread: number;
  name: string;
  usageCount: number;
  nodeIds: string[];
}

export interface ImageFillRef {
  imageRef: string;
  scaleMode: string;
  nodeId: string;
  nodeName: string;
}

export interface ComponentInventoryEntry {
  componentName: string;
  description?: string;
  variantProperties?: Record<string, string | boolean>;
  source: 'local' | 'library';
  usageCount: number;
}

  colors: ColorToken[];

  typography: TypographyToken[];
  spacing: SpacingToken[];

    Color extraction:

    - Two nodes with same solid fill color -> single color token with usageCount:2 and both nodeIds
    - Node with fills=[{type:'SOLID', color:{r:1,g:1,b:1,a:0.5}, visible:true}] -> colors contains {value:'rgba(255, 255, 255, 0.5)'}
    - Paint with visible:false -> NOT extracted (filtered out)
    - Node with strokes=[{type:'SOLID', color:{r:0,g:0,b:0,a:1}, visible:true}] -> color token with source includes 'stroke'
    - Same color from fill and stroke on different nodes -> single token, source has both 'fill' and 'stroke'

    Gradient extraction:
    - Node with fills=[{type:'GRADIENT_LINEAR', gradientHandlePositions:[...], gradientStops:[...], visible:true}] -> gradients has entry with CSS linear-gradient value

    Image fill extraction:
    - Node with fills=[{type:'IMAGE', scaleMode:'FILL', imageRef:'img:abc', visible:true}] -> imageFills has {imageRef:'img:abc', scaleMode:'FILL', nodeId, nodeName}
    - IMAGE fill does NOT produce a color token

    Typography extraction:
    - TEXT node with textStyle={fontFamily:'Inter', fontSize:16, fontWeight:600, lineHeightPx:24, letterSpacing:0} -> typography token created
    - Two TEXT nodes with identical textStyle -> single token with usageCount:2
    - TEXT node with different textStyle -> separate token
    - TEXT node with textStyleOverrides containing additional styles -> those also extracted as tokens
    - TEXT node with lineHeightPx undefined -> lineHeight is null

    Spacing extraction:
    - Node with autoLayout.padding={top:16, right:16, bottom:16, left:16} and gap=8 -> spacing tokens for 16 and 8
    - Symmetric padding -> single spacing token (not 4 duplicates)
    - Multiple nodes with same gap value -> single token with incremented usageCount
    - autoLayout.rowGap present -> also included as spacing token
    - Zero values (gap:0, padding:0) -> NOT included as spacing tokens

    Border extraction:
    - Node with cornerRadius=8, strokes=[{type:'SOLID', color:{r:0,g:0,b:0,a:1}}], strokeWeight=1 -> border token with radius:8, strokeColor:'#000000', strokeWeight:1
    - Node with rectangleCornerRadii=[4,4,8,8] -> border token with cornerRadii:[4,4,8,8], radius:null
    - Node with cornerRadius but no strokes -> border token with strokeColor:null, strokeWeight:null
    - Nodes with identical border properties -> deduplicated with usageCount

    Shadow extraction:
    - Node with effects=[{type:'DROP_SHADOW', visible:true, color:{r:0,g:0,b:0,a:0.25}, offset:{x:0,y:4}, radius:8, spread:0}] -> shadow token
    - Node with effects=[{type:'INNER_SHADOW', visible:true, ...}] -> shadow token with type:'inner-shadow'
    - Effect with visible:false -> NOT extracted
    - Non-shadow effects (LAYER_BLUR, BACKGROUND_BLUR) -> ignored
    - Two nodes with identical shadow -> single token with usageCount:2

    Component inventory:
    - INSTANCE node with componentRef -> component inventory entry with name, source, usageCount
    - Multiple INSTANCE nodes referencing same component -> single entry with usageCount summed (including repeatCount)
    - Component with description -> included in entry
    - Component with variantProperties -> included in entry
    - Local vs library components -> source tag correct

    Named style resolution:
    - Node with styleRefs={fill:'S:abc'} and stylesMap={'S:abc':{name:'Brand/Primary', styleType:'FILL'}} -> color token name is 'Brand/Primary'
    - Node without styleRefs -> color token gets auto-generated name (e.g., 'color-ff0000')
    - Node with styleRefs for text -> typography token gets style name

    Edge cases:
    - Empty tree (no nodes) -> all arrays empty
    - Node with empty fills=[] -> no color tokens extracted
    - Node with no style-related properties -> skipped cleanly
    - Deep nested tree (3+ levels) -> all descendant nodes processed

    **Step 2: GREEN — Implement src/tokens/collect.ts.**

    Core algorithm:
    1. Walk the LayoutNode tree recursively
    2. For each node, extract style data into token Maps keyed by canonical dedup key
    3. Resolve named styles from stylesMap using node.styleRefs
    4. After full walk, convert Maps to sorted arrays

    Implementation details:

    **Color extraction** from node.fills and node.strokes:
    - Filter visible paints: `paint.visible !== false`
    - For SOLID paints: combine paint.opacity (default 1) with color.a for effective alpha. Use figmaColorToCSS to get CSS value. Dedup key = the CSS value string.
    - For gradient paints: use gradientToCSS. Store as GradientToken.
    - For IMAGE paints: store as ImageFillRef with scaleMode and imageRef.
    - Track source ('fill' or 'stroke') on ColorToken.

    **Shadow extraction** from node.effects:
    - Filter: effect.visible === true AND (type === 'DROP_SHADOW' || type === 'INNER_SHADOW')
    - Convert effect.color using figmaColorToCSS
    - Dedup key: `${type}|${color}|${offsetX}|${offsetY}|${blur}|${spread}`
    - Also extract shadow colors as ColorTokens with source 'shadow'

    **Typography extraction** from node.textStyle (only TEXT nodes):
    - Extract fontFamily, fontSize, fontWeight, lineHeightPx (null if missing/undefined), letterSpacing
    - Dedup key: `${fontFamily}|${fontSize}|${fontWeight}|${lineHeight}|${letterSpacing}`
    - Also check node.textStyleOverrides: iterate values and extract each unique style

    **Spacing extraction** from node.autoLayout:
    - Collect all unique non-zero values from padding.top, padding.right, padding.bottom, padding.left, gap, rowGap
    - Use a Map<number, SpacingToken> keyed by numeric value
    - Track source labels: "padding-top", "padding-right", "gap", "row-gap", etc.

    **Border extraction**:
    - Nodes with cornerRadius OR rectangleCornerRadii OR (strokes with visible paints)
    - Build BorderToken with radius/cornerRadii, stroke color from first visible solid stroke, strokeWeight
    - Dedup key: `${radius}|${JSON.stringify(cornerRadii)}|${strokeColor}|${strokeWeight}`

    **Component inventory** from node.componentRef:
    - Key by componentId (from componentRef.componentId, available since Phase 2 capture but not on LayoutNode — use componentRef.componentName + variantProperties as the key)
    - Actually use `componentName + '::' + JSON.stringify(variantProperties)` as key since componentId is available on componentRef
    - Accumulate usageCount (including node.repeatCount ?? 1)
    - Include description, variantProperties, source from componentRef

    **Named style resolution:**
    - When extracting fills: check node.styleRefs?.fill against stylesMap
    - When extracting text: check node.styleRefs?.text against stylesMap
    - When extracting strokes: check node.styleRefs?.stroke against stylesMap
    - When extracting effects: check node.styleRefs?.effect against stylesMap
    - If style name found: use it as token name
    - If not found: auto-generate — colors: "color-{hex}", typography: "{family}-{size}-{weight}", borders: "border-{index}", shadows: "shadow-{index}", spacing: "spacing-{value}"

    **Sort output arrays:**
    - Colors: by usageCount descending (most used first)
    - Typography: by usageCount descending
    - Spacing: by value ascending (produces a scale: 4, 8, 12, 16...)
    - Borders: by usageCount descending
    - Shadows: by usageCount descending
    - Components: by usageCount descending

    Run tests — MUST pass.
