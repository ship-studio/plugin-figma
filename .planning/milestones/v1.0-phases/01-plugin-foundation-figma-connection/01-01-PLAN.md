    ```typescript
    export interface Shell {
      exec(command: string, args: string[], options?: { timeout?: number }): Promise<{

        stdout: string;
        stderr: string;
      }>;
    }

    export interface Storage {
      read(): Promise<Record<string, unknown>>;
      write(data: Record<string, unknown>): Promise<void>;
    }

    export interface PluginActions {
      showToast(message: string, type?: 'info' | 'success' | 'error'): void;
    }

    export interface PluginContextValue {
      project: { path: string; name: string };
      shell: Shell;
      storage: Storage;
      theme: { mode: 'light' | 'dark' };
      actions: PluginActions;
    }

    export interface FigmaUrlParts {
      fileKey: string;
      nodeId: string | null;
      fileType: 'file' | 'design' | 'proto' | 'board';
    }

    export type ExtractionScope = 'node' | 'frame' | 'page';

    export interface FigmaUser {
      id: string;
      handle: string;
      img_url: string;
      email?: string;
    }

    export interface FigmaFileInfo {
      name: string;
      pages: Array<{ id: string; name: string }>;
    }
    ```

    **Step 3: Create src/context.ts**

    Implement `usePluginContext()` hook that reads from `window.__SHIPSTUDIO_PLUGIN_CONTEXT_REF__`:
    ```typescript
    const React = (window as any).__SHIPSTUDIO_REACT__;
    const { useRef, useMemo } = React;

    export function usePluginContext(): PluginContextValue {
      const contextRef = (window as any).__SHIPSTUDIO_PLUGIN_CONTEXT_REF__;
      // contextRef.current provides the live context value
      return contextRef.current;
    }
    ```
    Note: Follow exact pattern from plugin-starter. The context ref is set by the host and always current.

    **Step 5: Create src/url-parser.ts**

    Implement `parseFigmaUrl(url: string): FigmaUrlParts | null` as a pure function.

    Use regex: `/figma\.com\/(file|design|proto|board)\/([A-Za-z0-9]+)/`
    Extract node-id from query string: `/[?&]node-id=([^&]+)/`
    Apply `decodeURIComponent()` then replace `-` with `:` for node ID conversion.
    Return null for non-matching URLs.

    Follow the exact implementation from 01-RESEARCH.md Pattern 2.

    **Step 7: Create src/figma-api.ts**

    Implement the Figma API client following 01-RESEARCH.md Pattern 1:
    - `figmaApiCall<T>(shell, endpoint, token, options?)`: Generic typed wrapper around shell.exec('curl', [...args]). Uses `-sS` flag, `X-Figma-Token` header, 120s default timeout. Parses JSON response, detects Figma error format ({ status, err }), throws typed errors for 403/404/429.
    - `validateToken(shell, token)`: Calls GET /v1/me, returns FigmaUser on success.
    - `validateFileAccess(shell, token, fileKey)`: Calls GET /v1/files/{fileKey}?depth=1, returns { name, pages } with page IDs and names.

    CRITICAL: Use `shell.exec('curl', ['-sS', '-H', \`X-Figma-Token: ${token}\`, url])` -- separate command and args array, NOT a single string.
    CRITICAL: Never include the token value in error messages.
