   ```typescript
   const handleNewBrief = useCallback(() => {
   ```

6. **Add view replacement conditional** -- Wrap the entire return JSX. When all result state is populated, render ResultsModal instead of the form:
   ```typescript
   if (briefResult && extractionResult && extractionStats && exportResult) {
     return (
       <ResultsModal
         briefResult={briefResult}
         extractionResult={extractionResult}
         extractionStats={extractionStats}
         exportResult={exportResult}
         onCopyBrief={handleCopyBrief}
         onNewBrief={handleNewBrief}
       />
     );
   ```
